# How to use battery

## create your app

```ts
export const App = () => {
        const battery = useBattery()
        return <span>{battery?.level}</span>
}
```

## create battery event

```ts
import { event } from 'reev'
import { useSyncExternalStore } from 'react'

export const self = event({
        async callback() {
                const snapshot = await navigator.getBattery()
                self.snapshot = snapshot
                self._callback()
        },
        mount() {
                self.callback()
                navigator.getBattery().then((target) => {
                        self.target = target
                        target.addEventListener('levelchange', self.callback)
                        target.addEventListener('chargingchange', self.callback)
                })
        },
        clean() {
                self.target.removeEventListener('levelchange', self.callback)
                self.target.removeEventListener('chargingchange', self.callback)
        },
})
```

## create useBattery hook

```ts
export const subscribe = (callback) => {
        self._callback = callback
        self.mount()
        return () => self.clean()
}

export const getSnapshot = () => self.snapshot

export const getServerSnapshot = () => void 0

export const useBattery = () => {
        return useSyncExternalStore(subscribe, getSnapshot, getServerSnapshot)
}
```

# How to use window size

## create your app

```ts
export const App = () => {
        const [width, height] = useWindowSize()
        return `${width} / ${height}`
}
```

## create window size event

```ts
import { event } from 'reev'
import { useSyncExternalStore } from 'react'

export const self = event({
        snapshot: [0, 0],
        mount() {
                window.addEventListener('resize', self.callback)
        },
        clean() {
                window.removeEventListener('resize', self.callback)
        },
        callback() {
                self.snapshot = [window.innerWidth, window.innerHeight]
        },
})
```

## create useWindowSize hook

```ts
export const subscribe = (callback) => {
        self({ callback })
        self.callback()
        self.mount()
        return () => self.clean()
}

export const getSnapshot = () => self.snapshot

export const getServerSnapshot = () => [0, 0]

export const useWindowSize = () => {
        return useSyncExternalStore(subscribe, getSnapshot, getServerSnapshot)
}
```

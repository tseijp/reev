# How to create resize event

## create your app

```tsx
import { gsap } from 'gsap'
import { useDrag } from 'reev/react'
//              or from '@reev/gesture/react'

export const App = () => {
        const { ref } = useResize(console.log)
        return <span ref={ref} style={style} />
}
```

### with Resize component

You can use shorthands Resize component

```jsx
import { Resize } from 'reev/react'

export const App2 = () => {
        return (
                <Resize onKey={console.log}>
                        {({ ref }) => <span ref={ref} style={style} />}
                </Resize>
        )
}
```

import { Resize } from 'reev/react'

export const App2 = () => {
        return (
                <Resize onKey={console.log}>
                        {({ ref }) => <span ref={ref} style={style} />}
                </Resize>
        )
}

### with Key component

You can use shorthands Key component

```jsx
import { Key } from 'reev/react'

export const App2 = () => {
        return (
                <Key onKey={console.log}>
                        {({ ref }) => <span ref={ref} style={style} />}
                </Key>
        )
}
```

import { Key } from 'reev/react'

export const App2 = () => {
        return (
                <Key onKey={console.log}>
                        {({ ref }) => <span ref={ref} style={style} />}
                </Key>
        )
}

<App2 />

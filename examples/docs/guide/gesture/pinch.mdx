# How to create pinch gesture

## create your app

```js
import { gsap } from 'gsap'
import { usePinch } from 'reev/react'

export const App = () => {
        const { ref } = usePinch(({ active, target, movement: [x, y] }) => {
                gsap.to(target, { scale: active ? 1.2 : 1, x, y })
        })
        return <span ref={ref} style={style} />
}
```

### with Pinch component

You can use shorthands Pinch component

```jsx
import { Pinch } from 'reev/react'

export const onPinch = ({ active, target, movement: [x, y] }) => {
        gsap.to(target, { x, y, scale: active ? 1.2 : 1 })
}

export const App2 = () => {
        return (
                <Pinch onPinch={onPinch}>
                        {({ ref }) => <span ref={ref} style={style} />}
                </Pinch>
        )
}
```

import { Pinch } from 'reev/react'
